55 void
56 MatlabManager::ProcessData( void* buffer, size_t size )
57 {
58  LOG_FUNCTION_START();
59  LogPrint( "Processing data of size: " , size );
60  if ( size <= 0 )
61  {
62  WarningPrint( "OpenDSS MATLAB Manager disconnected" );
63  delete this->m_client;
64  this->m_client = NULL;
65  LOG_FUNCTION_END();
66  return;
67  }
68 
69  int remainingSize = ( int )size;
70 
71  while ( remainingSize > 0 )
72  {
73 
74  TMessageType messageType;
75  char* currentAddress = ( char* )buffer;
76  memcpy( &messageType, currentAddress, sizeof( TMessageType ) );
77  currentAddress += sizeof( TMessageType );
78 
79  messageType = ntohl( messageType );
80 
81  if ( messageType == ClientCheckResultType )
82  {
83  remainingSize -= sizeof( TMessageType ) + sizeof( TClientCheckResult );
84  LogPrint( "Message Type: Client Presence Result" );
85  TClientCheckResult checkResult;
86  memcpy( &checkResult, currentAddress, sizeof( TClientCheckResult ) );
87  currentAddress += sizeof( TClientCheckResult );
88 
89  checkResult = ntohl( checkResult );
90  if ( checkResult == ClientExists )
91  {
92  LogPrint( "OpenDSS says: Client Exists" );
93  this->m_clientPresentInformation = true;
94  }
95  else
96  {
97  WarningPrint( "OpenDSS says: Client does not Exist" );
98  this->m_clientPresentInformation = false;
99  }
100  LogPrint( "Releasing Client Presence Semaphore" );
101  this->m_clientPresenceSemaphore.ReleaseSemaphore();
102  }
103  else if ( messageType == ClientWattageResultType )
104  {
105  remainingSize -= sizeof( TMessageType ) + sizeof( TWattage );
106  LogPrint( "Message Type: Client Wattage Result" );
107  TWattage wattageResult;
108  memcpy( &wattageResult, currentAddress, sizeof( TWattage ) );
109  currentAddress += sizeof( TWattage );
110 
111  wattageResult = ntohl( wattageResult );
112 
113  LogPrint( "OpenDSS says: Client Wattage = ", wattageResult );
114  this->m_clientWattageInformation = wattageResult;
115 
116  LogPrint( "Releasing Client Wattage Semaphore" );
117  this->m_clientWattageSemaphore.ReleaseSemaphore();
118  }
119  else if ( messageType == ClientVoltageResultType )
120  {
121  remainingSize -= sizeof( TMessageType ) + sizeof( TWattage );
122  LogPrint( "Message Type: Client Voltage Result" );
123  TVoltage voltageResult;
124  memcpy( &voltageResult, currentAddress, sizeof( TVoltage ) );
125  currentAddress += sizeof( TVoltage );
126 
127  voltageResult = ntohl( voltageResult );
128  this->m_clientVoltageInformation = voltageResult;
129  LogPrint( "OpenDSS says: Client Voltage = ", this->m_clientVoltageInformation );
130 
131  LogPrint( "Releasing Client Voltage Semaphore" );
132  this->m_clientVoltageSemaphore.ReleaseSemaphore();
133  }
134  else
135  {
136  ErrorPrint( "Unknwon Message Type from OPENDSS" );
137  }
138  buffer = currentAddress;
139  }
140  LOG_FUNCTION_END();
141 }

144 MatlabManager::IsClientPresent( const std::string & clientName )
145 {
146  LOG_FUNCTION_START();
147  LogPrint( "Checking with OpenDSS for client presence of: \"", clientName, "\"" );
148 
149  if ( this->m_client == NULL )
150  {
151  ErrorPrint( "No OpenDSS MATLAB connection" );
152  LOG_FUNCTION_END();
153  return ( -1 );
154  }
155 
156  this->m_clientPresentInformation = false;
157 
158  size_t bufferSize = sizeof(int) + sizeof(TMessageType) + clientName.length();
159  char* buffer = new char[bufferSize];
160  char* currentAddress = buffer;
161  int messageLength = htonl( bufferSize );
162 
163  memcpy( currentAddress, &messageLength, sizeof( int ) );
164  currentAddress += sizeof( int );
165 
166  TMessageType messageType = ClientCheckRequestType;
167  messageType = htonl( messageType );
168  memcpy( currentAddress, &messageType, sizeof( TMessageType ) );
169  currentAddress += sizeof( TMessageType );
170 
171  memcpy( currentAddress, clientName.c_str(), clientName.length() );
172  currentAddress += clientName.length();
173  LogPrint( "Sending Client Presence Message to OpenDSS" );
174  if ( this->m_client->SendData( buffer, bufferSize ) <= 0 )
175  {
176  ErrorPrint( "Send failed! OpenDSS MATLAB Connection is broken" );
177  delete[] buffer;
178  WarningPrint( "OpenDSS MATLAB Manager disconnected" );
179  delete this->m_client;
180  this->m_client = NULL;
181  LOG_FUNCTION_END();
182  return ( this->m_clientPresentInformation );
183  }
184  delete[] buffer;
185 
186  LogPrint( "Waiting for Client Presence Semaphore" );
187  this->m_clientPresenceSemaphore.TakeSemaphore();
188  LogPrint( "Client Presence Semaphore taken" );
189  LOG_FUNCTION_END();
190  return ( this->m_clientPresentInformation );
191 }
