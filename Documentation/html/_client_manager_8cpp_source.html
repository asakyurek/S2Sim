107 void
108 ClientManager::ProcessClientConnectionRequest( Asynchronous::ClientConnectionRequest* data )
109 {
110  LOG_FUNCTION_START();
111  LogPrint( "Asynchronous Client Connection Request received" );
112  Asynchronous::ClientConnectionResponse::TRequestResult requestResult;
113  if ( GetMatlabManager().IsClientPresent( data->GetClientName() ) )
114  {
115  LogPrint( "Client is present, accepting connection request" );
116  requestResult = Asynchronous::ClientConnectionResponse::RequestAccepted;
117  this->m_clientType = AsynchronousClient;
118  this->m_clientId = this->nextClientId;
119  LogPrint( "Registering client to the control manager" );
120  GetControlManager().RegisterClient( this->m_clientId, data->GetClientName(), this );
121  ++this->nextClientId;
122  }
123  else
124  {
125  WarningPrint( "Client is not present, rejecting connection request" );
126  requestResult = Asynchronous::ClientConnectionResponse::RequestObjectIdNotFound;
127  }
128  SystemManager::TSystemMode systemMode = GetSystemManager().GetSystemMode();
129  SystemManager::TSystemTime systemTime = GetSystemManager().GetSystemTime();
130  ConnectionManager::TNumberOfClients numberOfClients = GetConnectionManager().GetNumberOfClients();
131 
132  LogPrint( "Sending connection response message to client" );
133  Asynchronous::ClientConnectionResponse* responseData = Asynchronous::ClientConnectionResponse::GetNewClientConnectionResponse( 0x0000,
134  this->m_clientId,
135  requestResult,
136  systemTime,
137  numberOfClients,
138  systemMode );
139  this->m_client->SendData( responseData, ( int )Asynchronous::ClientConnectionResponse::GetSize() );
140  delete[] ( ( char* )responseData );
141  if ( requestResult == Asynchronous::ClientConnectionResponse::RequestObjectIdNotFound )
142  {
143  WarningPrint( "Deleting temporarily constructed client information for rejected request" );
144  GetConnectionManager().DeleteClient( this );
145  }
146  LOG_FUNCTION_END();
147 }
